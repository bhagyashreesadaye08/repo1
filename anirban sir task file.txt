alter procedure spget6
@purch_amt float,
@grade int
as
begin
 select c.customer_id,c.cust_name,o.salesman_id,o.ord_no,o.ord_date,s.name,o.purch_amt, c.grade from customer c join orders1 o on c.salesman_id=o.salesman_id 
 join salesman s on o.salesman_id=s.salesman_id where o.purch_amt>2000 and c.grade is not null 
end

alter procedure ghfdd
@cust_name varchar(50),
@ord_date date,
@grade int,
@name varchar(50),
@ord_no int,
@purch_amt float
as
begin
begin transaction
if exists(select*from recordsss where ord_no=@ord_no)
   update recordsss set cust_name=@cust_name,ord_date=@ord_date,grade=@grade,name=@name,purch_amt=@purch_amt where ord_no=@ord_no
 else
   insert into recordsss(cust_name,ord_date,grade,name,ord_no,purch_amt) values(@cust_name,@ord_date,@grade,@name,@ord_no,@purch_amt)
   commit transaction
   end
